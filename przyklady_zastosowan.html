new TypeIt("#przyklad1", {

3. Symulowanie używania backspace


new TypeIt("#przyklad3", {
    speed: 100,
    waitUntilVisible: true,
})
    .type("Przyy")
    .delete(1)
    .type("kład pokazujjaący")
    .move(-3)
    .delete(2)
    .move("END")
    .type(" efekt używamnia")
    .pause(200)
    .move(-3)
    .delete(1)
    .move("END")
    .type(" przycisku beckspace.")
    .pause(400)
    .move(-8)
    .delete(1)
    .type('a')
    .move("END")
    .go();
});

4. Pauzowanie i wznawianie animacji


let przyklad4_typeit = new TypeIt("#przyklad4", {
    strings: "Phasellus id luctus magna. Donec dignissim, ipsum quis vestibulum tristique, felis risus " +
    "viverra tortor, vel faucibus odio massa at tortor.",
    speed: 75,
    waitUntilVisible: true,
    loop: true,
}).go();
$('#btn-freeze').click(function () {
    przyklad4_typeit.freeze();
    $('#btn-freeze').hide();
    $('#btn-unfreeze').show();
});
$('#btn-unfreeze').click(function () {
    przyklad4_typeit.unfreeze();
    $('#btn-unfreeze').hide();
    $('#btn-freeze').show();
});

5. Kolorowy tekst


let przyklad5_typeit = new TypeIt('#przyklad5', {
    strings: ["Quisque feugiat ante eu arcu mattis porta. Sed a luctus est, vitae iaculis ex. " +
        "In sit amet nunc dignissim, gravida lectus vitae, molestie sem."],
    speed: 75,
    waitUntilVisible: true,
    afterStep: function(step, instance) {
        przyklad5_typeit.getElement().style.color = get_rand_color();
    }
}).go();
function get_rand_color()
{
    var color = Math.floor(Math.random() * Math.pow(256, 3)).toString(16);
    while(color.length < 6) {
        color = "0" + color;
    }
    return "#" + color;
}
